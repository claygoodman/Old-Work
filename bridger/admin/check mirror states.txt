SELECT db_name(database_id),mirroring_partner_instance, mirroring_witness_name, mirroring_state_desc, mirroring_role_desc,mirroring_safety_level, mirroring_witness_state_desc 
FROM sys.database_mirroring mirroring_quid IS NOT NULL
and (mirroring_state=1 OR mirroring_witness_state=2)

SELECT name, type_desc, state_desc, role_desc FROM sys.database_mirroring_endpoints
WHERE (state = 1 OR state = 2)
